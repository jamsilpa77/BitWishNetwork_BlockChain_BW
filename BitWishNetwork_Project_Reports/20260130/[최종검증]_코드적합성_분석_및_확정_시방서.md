### Step 1. [관리자] `server/routes/admin.ts`
*   **현재 상태:** `dummyReferrals` 배열 사용 (가짜 데이터).
*   **[확정 수정 코드]:**
    ```typescript
    import MiningState from '../models/MiningState'; // 임포트 확인됨
    import BonusRecord from '../models/BonusRecord'; // 임포트 확인됨

    // [GET] /api/admin/referral/all
    router.get('/referral/all', async (req, res) => {
        try {
            // 실제 BonusRecord 스키마 필드명(referralList, childWalletAddress) 검증 완료
            const records = await BonusRecord.aggregate([
                { $unwind: "$referralList" },
                {
                    $lookup: {
                        from: "miningstates", // MongoDB 컬렉션명 (소문자 복수형)
                        localField: "referralList.childWalletAddress",
                        foreignField: "walletAddress",
                        as: "miningInfo"
                    }
                },
                {
                    $project: {
                        referrerAddress: "$walletAddress",
                        referredAddress: "$referralList.childWalletAddress",
                        joinedDate: "$referralList.joinedAt",
                        kycStatus: "$referralList.isKycVerified",
                        isMining: { $arrayElemAt: ["$miningInfo.isMining", 0] }, // 배열 첫 요소 추출
                        accumulatedBonus: "$referralList.accumulatedBonus"
                    }
                },
                { $sort: { joinedDate: -1 } } // 최신순 정렬 추가
            ]);
            return res.json({ success: true, data: { records } });
        } catch (e) {
            return res.status(500).json({ success: false, message: (e as Error).message });
        }
    });
    ```

    public startSync() {
        if (this.syncIntervalInfo) clearInterval(this.syncIntervalInfo);
