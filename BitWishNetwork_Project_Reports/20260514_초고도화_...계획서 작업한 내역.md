import './FloatingArchiveWindow.css';

interface NoticeItem {
    _id: string;
    title: string;
    createdAt: string;
}

interface Props {
    onClose: () => void;
}

const FloatingArchiveWindow: React.FC<Props> = ({ onClose }) => {
    const [notices, setNotices] = useState<NoticeItem[]>([]);
    const [page, setPage] = useState(1);
    const [hasMore, setHasMore] = useState(true);
    const [isLoading, setIsLoading] = useState(false);
    
    // 드래그 상태 관리
    const [position, setPosition] = useState({ x: 100, y: 150 });
    const isDragging = useRef(false);
    const offset = useRef({ x: 0, y: 0 });
    
    const observer = useRef<IntersectionObserver | null>(null);
    const lastElementRef = useCallback((node: HTMLDivElement | null) => {
        if (isLoading) return;
        if (observer.current) observer.current.disconnect();
        observer.current = new IntersectionObserver(entries => {
            if (entries[0].isIntersecting && hasMore) {
                setPage(prev => prev + 1);
            }
        });
        if (node) observer.current.observe(node);
    }, [isLoading, hasMore]);

    // 다국어 지원 (절대 규칙 준수)
    const [lang] = useState(() => localStorage.getItem('bw_lang') || 'ko');
    const dict: any = {
        ko: { title: '비트위시 공지 아카이브', close: '닫기' },
        en: { title: 'BitWish Notice Archive', close: 'Close' },
        ja: { title: 'お知らせアーカイブ', close: '閉じる' },
        zh: { title: '公告存档', close: '关闭' }
    };
    const t = (key: string) => dict[lang]?.[key] || dict['ko'][key];

    // 데이터 로딩
    const fetchNotices = useCallback(async () => {
        if (isLoading) return;
        setIsLoading(true);
        try {
            const res = await fetch(`http://localhost:5001/api/notices/archive?page=${page}&limit=10`);
            const data = await res.json();
            if (data.success) {
                setNotices(prev => [...prev, ...data.data]);
                setHasMore(data.data.length === 10);
            }
        } catch (error) {
            console.error('Archive fetch failed:', error);
        } finally {
            setIsLoading(false);
        }
    }, [page]);

    useEffect(() => {
        fetchNotices();
    }, [fetchNotices]);

    // 드래그 로직 (Step 5-1)
    const handleMouseDown = (e: React.MouseEvent) => {
        isDragging.current = true;
        offset.current = {
            x: e.clientX - position.x,
            y: e.clientY - position.y
        };
        document.addEventListener('mousemove', handleMouseMove);
        document.addEventListener('mouseup', handleMouseUp);
    };

    const handleMouseMove = (e: MouseEvent) => {
        if (!isDragging.current) return;
        setPosition({
            x: e.clientX - offset.current.x,
            y: e.clientY - offset.current.y
        });
    };

    const handleMouseUp = () => {
        isDragging.current = false;
        document.removeEventListener('mousemove', handleMouseMove);
        document.removeEventListener('mouseup', handleMouseUp);
    };

                <button className="header-close" onClick={onClose}>&times;</button>
            </div>

                    {notices.map((notice, index) => {

                                    <span className="item-title">{notice.title}</span>
                                    <span className="item-date">{new Date(notice.createdAt).toLocaleDateString()}</span>
                                </div>
                            );

                                <span className="item-title">{notice.title}</span>
                                <span className="item-date">{new Date(notice.createdAt).toLocaleDateString()}</span>
                            </div>
                        );
                    })}

                </div>
            </div>
            
            <div className="bw-archive-footer">
                <button className="footer-close-btn" onClick={onClose}>{t('close')}</button>
            </div>
        </div>
    );
};
