```typescript
/**
 * 특정 사용자의 출석 데이터 초기화
 * @param userId 사용자 ID (지갑 주소)
 * @returns 초기화 결과
 */
public resetUserData(userId: string): {
    success: boolean;
    message: string;
} {
    try {
        let deletedCount = 0;

        // 1. 사용자 출석 기록 삭제
        if (this.userAttendanceRecords.has(userId)) {
            this.userAttendanceRecords.delete(userId);
            deletedCount++;
        }

        // 2. 사용자 출석 상태 삭제
        if (this.userAttendanceStatus.has(userId)) {
            this.userAttendanceStatus.delete(userId);
            deletedCount++;
        }

        // 3. 출석 보너스 보관함 삭제
        if (this.attendanceBonusStorage.has(userId)) {
            this.attendanceBonusStorage.delete(userId);
            deletedCount++;
        }

        // 4. localStorage에 저장
        this.saveData();

        console.log(`[AttendanceBonusService] 사용자 데이터 초기화 완료: ${userId}, 삭제 항목: ${deletedCount}개`);

        return {
            success: true,
            message: `출석 데이터 초기화 완료 (${deletedCount}개 항목 삭제)`
        };
    } catch (error) {
        console.error('[AttendanceBonusService] 사용자 데이터 초기화 실패:', error);
        return {
            success: false,
            message: '초기화 실패'
        };
    }
}
```
